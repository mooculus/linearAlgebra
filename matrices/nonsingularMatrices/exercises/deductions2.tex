\begin{exercise}
  A certain nonhomogeneous system of linear equations has a
  coefficient matrix $A$ which is nonsingular.
  
  What could be true about this system's solution set?  Select all that
  are possible.
  \begin{selectAll}
    \choice{There are no solutions.}
    \choice[correct]{There is exactly one solution.}
    \choice{There are infinitely many solutions.}
  \end{selectAll}
  
  Is the zero vector a solution to this system?
  \begin{multipleChoice}
    \choice[correct]{No, the zero vector is not a solution.}
    \choice{Yes, the zero vector is a solution.}
    \choice{It depends on the system.}
  \end{multipleChoice}
\end{exercise}
